
● pytest でのメソッド別テスト実行方法を説明します：

  基本的な実行方法

  1. 特定のテストメソッド

  # パターン: ファイル::クラス::メソッド
  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py::TestLlmConfig::test_valid_config -v

  2. 特定のテストクラス

  # クラス全体のテスト
  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py::TestLlmConfig -v

  3. ファイル全体

  # ファイル全体
  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py -v

  便利なオプション

  詳細表示

  # 詳細情報付き
  pytest trail_status/tests/test_llm_client.py::TestLlmConfig::test_valid_config -v -s

  パターンマッチ

  # メソッド名でパターンマッチ
  pytest -k "test_api_key" -v

  # 複数パターン
  pytest -k "test_api_key or test_valid" -v

  # クラス名でパターンマッチ
  pytest -k "TestLlmConfig" -v

  失敗時に即停止

  pytest trail_status/tests/test_llm_client.py -x -v

  実際の例

  API keyテストのみ

  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py -k "api_key" -v

  非同期テストのみ

  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py -k "asyncio" -v

  LlmConfigテストのみ

  docker compose exec web uv run pytest trail_status/tests/test_llm_client.py::TestLlmConfig -v

  デバッグ時の実行

  # print文も表示
  pytest trail_status/tests/test_llm_client.py::TestLlmConfig::test_valid_config -v -s

  # pdbデバッガー起動
  pytest trail_status/tests/test_llm_client.py::TestLlmConfig::test_valid_config --pdb
